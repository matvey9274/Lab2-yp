#include <iostream>

using namespace std;

int main()
{
	setlocale( LC_ALL,"Russian" );
	int n,m;
	cout << "Количество участников n = ";
	cin >> n;
	cout << "Количество выстрелов m = ";
	cin >> m;

	// выделяем память под массив выстрелов
	int **A;
	A = new int* [n]; // n строчек: по одной на каждого стрелка
	for(int i=0; i<n; i++)
		A[i] = new int[m]; // m столбцов: по одному на выстрел
	// выделяем память под массив лучших выстрелов: по одному элементу на стрелка
	int *M;
	M = new int[n];
	// выделяем память под массив суммы очков за все выстрелы: по одному элементу на стрелка
	int *S;
	S = new int[n];

	// в цикле перебираем номера стрелков
	for(int i=0; i<n; i++){
		cout << "Стрелок " << i+1 << endl;
		M[i] = 0; // в начале лучший выстрел = 0
		S[i] = 0; // в начале сумма очков = 0
		// цикл в цикле: перебираем для данного стрелка номера выстрелов
		for(int j=0; j<m; j++){
			cout << "Выстрел " << j+1 <<": ";
			cin >> A[i][j]; // вводим количество очков за этот выстрел
			S[i] += A[i][j]; // прибавляем количество очков к сумме
			if(A[i][j] > M[i]) // проверяем, был ли этот выстрел наилучшим
				M[i] = A[i][j]; // если да, перезаписываем лучший балл
		}
		cout << endl;
	}

	// ищем наибольшие баллы
	int max1 = 0; // количество баллов за лучший выстрел в состязании, в начале = 0
	int max2 = 0; // наибольшая сумма баллов у стрелков, сделавших лучший выстрел, в начале = 0

	for(int i=0; i<n; i++) // перебираем всех стрелков
		if(M[i]>max1) // если его лучший выстрел больше по баллам среди всех
			max1 = M[i]; // перезаписываем максимальный балл в состязании

	for(int i=0; i<n; i++) // перебираем всех стрелков
		if(M[i]==max1)	// если этот стрелок набрал максимальный балл в состязании
			if(S[i]>max2) // если сумма его баллов наибольшая
				max2 = S[i]; // перезаписываем наибольшую сумму баллов

	cout << "Победители: " << endl;
	for(int i=0; i<n; i++) // перебираем всех стрелков
		if((M[i]==max1) && (S[i]==max2)) // если этот стрелок набрал максимальный балл за выстрел и максимальную сумму среди всех
			cout << "Стрелок " << i+1 << endl; // выводим его в списке победителей

	// освобождаем выделенную память
	delete [] S;
	delete [] M;
	for(int i=0; i<n; i++)
		delete [] A[i];
	delete [] A;
	return 0;
}
